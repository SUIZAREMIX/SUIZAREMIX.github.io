Snippet Funcional: Factory Pattern y SOLID en Spring Boot
// Aplicando Factory Pattern y Principio Open/Closed (SOLID)
@Component
public class PaymentFactory {
    private final Map<String, PaymentProcessor> strategies = new HashMap<>();

    @Autowired
    public PaymentFactory(List<PaymentProcessor> processors) {
        processors.forEach(p -> strategies.put(p.getClass().getSimpleName(), p));
    }

    public PaymentProcessor getProcessor(String type) {
        return Optional.ofNullable(strategies.get(type))
            .orElseThrow(() -> new IllegalArgumentException("Procesador no soportado"));
    }
}

Snippet Funcional: API Asíncrona con FastAPI y Pandas
from fastapi import FastAPI, BackgroundTasks
import pandas as pd
import asyncio

app = FastAPI()

async def process_data_pipeline(file_path: str):
    await asyncio.sleep(1) # Simulando I/O asíncrono
    df = pd.read_csv(file_path).dropna()
    metrics = df.groupby('category').agg({'revenue': 'sum'}).sort_values('revenue', ascending=False)
    metrics.to_json(f"{file_path}_processed.json")

@app.post("/api/v1/analyze")
async def analyze_dataset(file_path: str, bg_tasks: BackgroundTasks):
    bg_tasks.add_task(process_data_pipeline, file_path)
    return {"status": "processing", "message": "Pipeline iniciado en background"}

Snippet Funcional: Algoritmo de Dijkstra (Estructuras Eficientes) 
import heapq

def shortest_path(graph, start, end):
    # graph: {nodo: {vecino: peso}}
    distances = {node: float('infinity') for node in graph}
    distances[start] = 0
    pq = [(0, start)] # (distancia_acumulada, nodo_actual)
    
    while pq:
        current_dist, current_node = heapq.heappop(pq)
        if current_node == end: return current_dist
        if current_dist > distances[current_node]: continue
            
        for neighbor, weight in graph[current_node].items():
            dist = current_dist + weight
            if dist < distances[neighbor]:
                distances[neighbor] = dist
                heapq.heappush(pq, (dist, neighbor))
    return -1 # Ruta no encontrada
